        self.pacientes = []

    def medico_disponivel(self, nome, data, hora):

            return False

        return True

    def agendar_consulta(self, paciente, medico, data, hora):
        if paciente not in self.pacientes:

        if not self.medico_disponivel(medico, data, hora):
